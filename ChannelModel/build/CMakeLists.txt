cmake_minimum_required(VERSION 3.5)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED True)
set(pybind11_DIR "venv/Lib/site-packages/pybind11/share/cmake/pybind11")
project(example VERSION 1.0)
find_package(pybind11 REQUIRED)
pybind11_add_module(Channel channel.cpp)
pybind11_add_module(example example.cpp)
target_include_directories(Channel PUBLIC "${PROJECT_BINARY_DIR}")
target_include_directories(example PUBLIC "${PROJECT_BINARY_DIR}")
